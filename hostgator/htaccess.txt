# Caso o HostGator não permita arquivos iniciados com ponto, mantenha este arquivo
# como `htaccess.txt`. Se for possível renomear, troque para `.htaccess`.
#
# Define ads.html como entrada padrão e evita reescrever arquivos reais.
DirectoryIndex ads.html

<IfModule mod_rewrite.c>
  RewriteEngine On
  RewriteBase /
  RewriteRule ^ads\.html$ - [L]
  RewriteRule ^ads-assets/ - [L]
  RewriteRule ^tracking\.js$ - [L]
  RewriteRule ^01\.png$ - [L]
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteRule . /ads.html [L]
</IfModule>
