# Arquivo de configuração para hospedagem em Apache/HostGator.
# Garante que a landing page utilize `ads.html` como entrada padrão.
DirectoryIndex ads.html

<IfModule mod_rewrite.c>
  RewriteEngine On
  RewriteBase /
  # URLs limpas por canal (preserva UTMs e outros params).
  RewriteRule ^youtube/?$ /?channel=youtube [L,QSA]
  RewriteRule ^instagram/?$ /?channel=instagram [L,QSA]
  RewriteRule ^ads\.html$ - [L]
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteRule . /ads.html [L]
</IfModule>
